```typescript
  error?: string;
}
```

**onSession 回调签名（用于 GroupQueue.registerSession()）：**
```typescript
type OnSession = (session: SDKSession) => void;
```

**runInProcessAgent 实现要点：**

1. 从 `.env` 读取 `ANTHROPIC_API_KEY`（用 `readEnvFile`），优先级：`.env` > `process.env`
2. 按规则确定 `groupCwd`：
   - `isMain` 为 true → `process.cwd()`（项目根目录）
   - 否则 → `path.resolve(GROUPS_DIR, group.folder)`
3. `sessionStorePath` = `path.resolve(DATA_DIR, 'sessions', group.folder)`
4. `sessionOptions` 包含：
   - `model`: `process.env.CLAUDE_MODEL || 'claude-opus-4-6'`
   - `cwd`: groupCwd
   - `permissionMode: 'bypassPermissions'`
   - `allowDangerouslySkipPermissions: true`
   - `settingSources: ['project']`（读取各 cwd 下的 CLAUDE.md）
   - `env`: `{ ...process.env, ANTHROPIC_API_KEY: apiKey, CLAUDE_HOME: sessionStorePath }`
5. 根据 `input.sessionId` 选择 `unstable_v2_resumeSession` 或 `unstable_v2_createSession`（不使用 `await using`）
6. 调用 `onSession(session)`（让调用方拿到 session 句柄存入 GroupQueue）
7. `await session.send(input.prompt)`
8. 迭代 `session.stream()`：
   - 每条消息都更新 `capturedSessionId = msg.session_id`
   - `msg.type === 'assistant'`：提取所有 `type === 'text'` 的 content blocks 拼接，非空则调用 `onOutput({ status: 'success', result: text, newSessionId: capturedSessionId })`，设 `hadOutput = true`
   - `msg.type === 'result'`：
     - `msg.subtype === 'success'` → 调用 `onOutput({ status: 'success', result: null, newSessionId: capturedSessionId })`（idle signal）
     - 否则 → 记录 `errorSubtype = msg.subtype`，调用 `onOutput({ status: 'error', result: null, error: msg.subtype })`
9. stream 异常：catch 后返回 `{ status: 'error', result: null, error: errMsg }`（用 pino logger 记录）
10. 正常返回：`{ status: errorSubtype ? 'error' : 'success', result: null, newSessionId: capturedSessionId, error: errorSubtype }`

**注意事项（来自 RESEARCH.md）：**
- 不使用 `await using`，避免 session 在函数退出时自动关闭，阻断 follow-up turns
- `env` 选项传 API key，不修改 `process.env`（避免并发竞争）
- 不设 `allowedTools`（与容器模式行为等价，后续可收紧）
- pino 日志使用 `logger.error({ group: group.name, error }, 'SDK session error')` 模式
