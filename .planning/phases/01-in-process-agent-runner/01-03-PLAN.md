  registeredGroups: () => Record<string, RegisteredGroup>;

**3. 替换 runTask() 内部的 runContainerAgent 调用：**

删除：
```typescript
const TASK_CLOSE_DELAY_MS = 10000;
let closeTimer: ReturnType<typeof setTimeout> | null = null;
const scheduleClose = () => { ... };

const output = await runContainerAgent(
  group,
  { ... },
  (proc, containerName) =>
    deps.onProcess(task.chat_jid, proc, containerName, task.group_folder),
  async (streamedOutput: ContainerOutput) => { ... },
);

if (closeTimer) clearTimeout(closeTimer);
```

替换为：
```typescript
const output = await runInProcessAgent(
  group,
  {
    prompt: task.prompt,
    sessionId,
    groupFolder: task.group_folder,
    chatJid: task.chat_jid,
    isMain,
    isScheduledTask: true,
    assistantName: ASSISTANT_NAME,
  },
  (session) => deps.onSession(task.chat_jid, session, task.group_folder),
  async (streamedOutput: AgentOutput) => {
    if (streamedOutput.result) {
      result = streamedOutput.result;
      await deps.sendMessage(task.chat_jid, streamedOutput.result);
    }
    if (streamedOutput.status === 'success') {
      deps.queue.notifyIdle(task.chat_jid);
    }
    if (streamedOutput.status === 'error') {
      error = streamedOutput.error || 'Unknown error';
    }
  },
);
```

注意：删除 scheduleClose / TASK_CLOSE_DELAY_MS / closeTimer 逻辑——V2 session 的 turn 结束后 stream() 自然返回，不需要靠定时器关闭。如果需要任务完成后 session 关闭，依赖 notifyIdle → closeStdin（已改为 session.close()）。

**4. 修改 output 处理部分（output.status 检查）：**
